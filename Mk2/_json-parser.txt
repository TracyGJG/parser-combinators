Parsers
========

Value
  SEQUENCE
    OPTIONAL WhiteSpace
    CHOICE
      OPTIONAL String
      OPTIONAL Number
      OPTIONAL Array
      OPTIONAL Object
      OPTIONAL Literal
    OPTIONAL WhiteSpace

Array
  SEQUENCE
    REQUIRED ArrayStart(1)
    CHOICE
      OPTIONAL WhiteSpace
      SEQUENCE
        REQUIRED Value
        MULTIPLE
          SEQUENCE
            REQUIRED ValueSeparater(1)
            REQUIRED Value
    REQUIRED ArrayStart(1)

Number
  SEQUENCE
    REQUIRED Integer
    OPTIONAL Fraction
    OPTIONAL Exponent

Object
  SEQUENCE
    REQUIRED ObjectStart(1)
    CHOICE
      OPTIONAL WhiteSpace
      SEQUENCE
        REQUIRED Property
        MULTIPLE
          SEQUENCE
            REQUIRED ValueSeparater(1)
            REQUIRED Property
    REQUIRED ObjectEnd(1)

String
  SEQUENCE
    REQUIRED StringDelim(1)
    MULTIPLE
      OPTIONAL PrinatableChar(1)
      OPTIONAL EscapeChar(2)
      OPTIONAL UnicodeChar(6)
    REQUIRED StringDelim(1)

WhiteSpace
  MULTIPLE
    OPTIONAL WhiteSpaceChar(1)

----

COMBINATORS
============

SEQUENCE:: [parsers] -> (state, index) -> state // All parsers advance, or EOI.
CHOICE:: [parsers] -> (state, index) -> state // One parser advances, or EOI.
MULTIPLE:: [parsers] -> state -> state // Repeat zero or more until fail or EOI.
REQUIRED:: (parser, error) -> (state, index) -> state // parser advances, or EOI.

----

Literal
  CHOICE
    OPTIONAL True(1)
    OPTIONAL False(1)
    OPTIONAL Null(1)

Property
  SEQUENCE
    OPTIONAL WhiteSpace
    REQUIRED String
    OPTIONAL WhiteSpace
    REQUIRED KeySeparater(1)
    REQUIRED Value

Integer
  SEQUENCE
    OPTIONAL Negative(1)
    CHOICE
      OPTIONAL Zero(1)
      SEQUENCE
        REQUIRED LeadingDigit(1)
        MULTIPLE Digit(1)

Fraction
  SEQUENCE
    REQUIRED DecimalPoint(1)
    REQUIRED Digit(1)
    MULTIPLE Digit(1)

Exponent
  SEQUENCE
    REQUIRED ExponentChar(1)
    OPTIONAL SignChar(1)
    REQUIRED Digit(1)
    MULTIPLE Digit(1)
